204    * Does the login
205    * 
206    * @throws Exception
207    */
208   public void login() throws Exception {
209     if (!loggedIn) {
210       if (restricted) {
211         wikiUser = getUser();
212         if (wikiUser == null) {
213           throw new IllegalStateException("backend user not configured");
214         }
215         auth = login(wikiUser.getUsername(), wikiUser.getPassword());
216       }
217       loggedIn = true;
218     }
219   }

325   /**
326    * check the category
327    * 
328    * @param content
329    * @param categoryName
330    * @return
331    * @throws Exception
332    */
333   public boolean checkCategory(String content, String categoryName)
334       throws Exception {
335     boolean result = false;
336     if (content != null) {
337       String pattern = "(\\[\\[Category:)(.*?)(\\]\\])";
338       for (Matcher m = Pattern.compile(pattern, Pattern.MULTILINE)
339           .matcher(content); m.find();) {
340         String category = m.group(2).trim();
341         if (category.equals(categoryName.trim())) {
342           result = true;
343           break;
344         }
345       }
346     }
347     return result;
348   }

363   /**
364    * fix the given attribute of the given node if the value starts with the given prefix
365    * e.g. in an img node the src attribute might be fixed if it starts with "/images"
366    * @param node
367    * @param attribute
368    * @param prefix
369    * @return the original attribute Value
370    */
371   public String fixNode(TagNode node, String attribute, String prefix) {
372     String attributeValue = node.getAttributeByName(attribute);
373     if (attributeValue != null && attributeValue.startsWith(prefix)) {
374       node.removeAttribute(attribute);
375       node.addAttribute(attribute, this.getBaseUrl() + attributeValue);
376     }
377     return attributeValue;
378   }

380   /**
381    * make sure img tags with src pointing to local /images are rewritten to
382    * point to original
383    * 
384    * @param root
385    * @throws Exception
386    */
387   public void fixImages(TagNode root) throws Exception {
388     // fix images
389     List<TagNode> imgNodes = Html.getXpathNodes(root, "//img");
390     for (TagNode img : imgNodes) {
391       String src=fixNode(img,"src","/images");
392       String srcset = img.getAttributeByName("srcset");
393       if (srcset != null) {
394         String newsrcset = "";
395         String delim = "";
396         String[] srcsetparts = srcset.split(",");
397         for (String srcsetpart : srcsetparts) {
398           String newsrcsetpart = srcsetpart.trim();
399           if (newsrcsetpart.startsWith("/images")) {
400             newsrcsetpart = this.getBaseUrl() + srcsetpart;
401           }
402           newsrcset += delim + newsrcsetpart;
403           delim = ", ";
404         }
405         img.removeAttribute("srcset");
406         img.addAttribute("srcset", newsrcset);
407       }
408       if (debug) {
409         LOGGER.log(Level.INFO, "src=" + src);
410       }
411     }
412   }

427   /**
428    * fix object tags e.g for PDF display
429    * 
430    * @param root
431    * @throws Exception
432    */
433   public void fixObjects(TagNode root) throws Exception {
434     List<TagNode> objectNodes = Html.getXpathNodes(root, "//object");
435     for (TagNode objectNode : objectNodes) {
436       fixNode(objectNode,"data","/images");
437       List<TagNode> anchorNodes = Html.getXpathNodes(objectNode, "//a");
438       for (TagNode anchorNode : anchorNodes) {
439         fixNode(anchorNode,"href","/images");
440       }
441     }
442   }

458   /**
459    * remove the nodes with the given tagname
460    * 
461    * @param root
462    * @param tagName
463    */
464   public void removeTagNode(TagNode root, String tagName) {
465     TagNode[] tags = root.getElementsByName(tagName, false);
466     if (tags != null && tags.length == 1) {
467       TagNode tag = tags[0];
468       TagNode parent = tag.getParent();
469       List<TagNode> childs = tag.getChildTagList();
470       // remove the full node
471       parent.removeChild(tag);
472       if ("body".equals(tagName)) {
473         for (TagNode child : childs) {
474           parent.addChild(child);
475         }
476       }
477     }
478   }

480   /**
481    * fix the MediaWiki elements in the given html code
482    * 
483    * @param html
484    *          - the original html
485    * @return - the html with fixes
486    * @throws Exception
487    */
488   public String fixMediaWikiHtml(String html) throws Exception {
489     TagNode root = Html.getDom(html);
490     fixImages(root);
491     fixVideos(root);
492     fixObjects(root);
493     removeEditSections(root);
494     this.removeTagNode(root, "head");
495     this.removeTagNode(root, "body");
496     html = Html.getHtml(root);
497     if (html == null)
498       return "";
499     return html;
500   }

502   /**
503    * frame the html retrieved the given title using the rythm template for this
504    * BackendWiki
505    * 
506    * @param sitePage - the page to frame
507    * @param uriInfo 
508    * @return the html code
509    * @throws Exception
510    */
511   public String frame(SitePageInfo sitePage, UriInfo uriInfo) throws Exception {
512     Map<String, Object> rootMap = new HashMap<String, Object>();
513     rootMap.put("uriInfo", uriInfo);
514     rootMap.put("postService", PostManager.getInstance());
515     if (sitePage.getPostToken()!=null) {
516       rootMap.put("postToken", sitePage.getPostToken());
517     }
518     rootMap.put("title", sitePage.getPageTitle());
519     if (siteinfo != null)
520       rootMap.put("lang", this.siteinfo.getLang());
521     String html = getPageHtml(sitePage.getPageTitle());
522     html = fixMediaWikiHtml(html);
523     rootMap.put("content", html);
524     String lframe=this.getFrame();
525     if (this.smw) {
526       PropertyMap smwprops = this.getSMWProperties(sitePage.getPageTitle());
527       if (smwprops.getMap().containsKey("Frame"))
528         lframe=smwprops.get("Frame");
529       rootMap.put("smwprops", smwprops);
530     }
531     String template = getTemplate(lframe);
532     RythmContext rythmContext = RythmContext.getInstance();
533     if (this.wikiTemplateResourceLoader==null) {
534       wikiTemplateResourceLoader=new WikiTemplateResourceLoader(this,rythmContext);
535     }
536     String result = rythmContext.render(template, rootMap);
537     return result;
538   }

540   private static final String SOURCETAG_BEGIN = "<source";
541   private static final String SOURCE_ATTRIBUTE_PATTERN = "[\\s](id|lang|cache|line)(\\s?=?\'([^\']*)\')?";
542   private static final String SOURCETAG_PATTERN = SOURCETAG_BEGIN + "("
543       + SOURCE_ATTRIBUTE_PATTERN + ")*" + "[\\s]?>";
544   static final Pattern SOURCETAG_REGEX = Pattern.compile(SOURCETAG_PATTERN);
545   private static final String SOURCETAG_END = "</source>";
546 
547   /**
548    * get the template Code from the given pageTitle
549    * 
550    * @param pageTitle
551    * @return the templateCode
552    * @throws Exception
553    */
554   public String getTemplate(String pageTitle) throws Exception {
555     String pageContent = this.getPageContent(pageTitle);
556     if ((pageContent == null) || pageContent.isEmpty()) {
557       pageContent = "error: template " + pageTitle + " is empty";
558     } else {
559       final Matcher matcher = SOURCETAG_REGEX.matcher(pageContent);
560       if (matcher.find()) {
561         int matchend = matcher.end();
562         int sourcendpos = pageContent.indexOf(SOURCETAG_END);
563         if (sourcendpos > 0) {
564           pageContent = pageContent.substring(matchend, sourcendpos);
565         }
566       }
567     }
568     return pageContent;
569   }

571   /**
572    * get the semantic MediaWiki properties for the given pageTitle
573    * 
574    * @param pageTitle
575    *          - the pageTitle to get the properties for
576    * @return - a map of properties
577    * @throws Exception
578    */
579   public PropertyMap getSMWProperties(String pageTitle)
580       throws Exception {
581     PropertyMap result=new PropertyMap();
582     String params = "&subject=" + super.encode(pageTitle);
583     Api api = getActionResult("browsebysubject", params, null, null, "json");
584     if (api != null) {
585       Query query = api.getQuery();
586       List<Property> data = query.getData();
587       result.init(data);
588     }
589     return result;
590   }
