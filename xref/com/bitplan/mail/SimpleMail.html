21  package com.bitplan.mail;
22  
23  import java.io.File;
24  import java.io.FileReader;
25  import java.util.Properties;
26  
27  import org.simplejavamail.mailer.Mailer;
28  
29  import com.bitplan.wikifrontend.BackendWiki;
30  
31  public class SimpleMail {
32  
33    /**
34     * get a mailer initialized from the mail.ini configuration file in the
35     * .wikibackend directory of the user running this Server
36     * 
37     * @return - the Mailer
38     * @throws Exception
39     */
40    public static Mailer getMailer() throws Exception {
41      File propertyDir = BackendWiki.getPropertyDir();
42      File propertyFile = new File(propertyDir, "mail.ini");
43      if (propertyFile.exists()) {
44        Properties props = new Properties();
45        props.load(new FileReader(propertyFile));
46        String server = props.getProperty("mail.server");
47        String username = props.getProperty("mail.username");
48        int port = Integer.parseInt(props.getProperty("mail.port", "25"));
49        String password = props.getProperty("mail.password");
50        Mailer mailer = new Mailer(server, port, username, password);
51        return mailer;
52      } else
53        return null;
54    }
55  }
