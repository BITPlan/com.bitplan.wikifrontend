42  /**
43   * loader for rythm templates from a wiki
44   * @author wf
45   *
46   */
47  public class WikiTemplateResourceLoader implements ITemplateResourceLoader {
48    protected static Logger LOGGER = Logger.getLogger("com.bitplan.rythm");
49    
50    private RythmEngine engine;
51    private Map<String, WikiTemplateResource> resources = new LinkedHashMap<String, WikiTemplateResource>();
52    private BackendWiki wiki;
53    
54    /**
55     * construct me for the given wiki
56     * 
57     * @param wiki
58     * @param rythmContext 
59     */
60    public WikiTemplateResourceLoader(BackendWiki wiki, RythmContext rythmContext) {
61      setWiki(wiki);
62      setEngine(rythmContext.getEngine());
63      getEngine().registerResourceLoader(this);
64    }
65  
66    @Override
67    public void setEngine(RythmEngine engine) {
68      this.engine = engine;
69    }
70  
71    @Override
72    public RythmEngine getEngine() {
73      return this.engine;
74    }
75    
76    public BackendWiki getWiki() {
77      return wiki;
78    }
79  
80    public void setWiki(BackendWiki wiki) {
81      this.wiki = wiki;
82    }
83    
84    /**
85     * get a stack trace for the given throwable
86     * 
87     * @param t
88     *          - the throwable to get the stack trace for
89     * @return - the stacktrace as a string
90     */
91    public String getStackTrace(Throwable t) {
92      StringWriter sw = new StringWriter();
93      t.printStackTrace(new PrintWriter(sw));
94      return sw.toString();
95    }
96  
97    /**
98     * get an error message for the given throwable
99     * 
100    * @param at
101    *          - the point where the error happened e.g. some method
102    * @param th
103    *          - the throwable to get the error message for
104    * @return - the error Message
105    */
106   public String getErrMessage(String at, Throwable th) {
107     String result = "\nError at " + at + ": " + th.getClass().getName() + "\n";
108     result += th.getMessage() + "\n";
109     result += getStackTrace(th);
110     return result;
111   }
112 
113   /**
114    * handle the given Exception from the given method
115    * 
116    * @param method
117    * @param th
118    * @return the error Message
119    */
120   public void handle(String method, Throwable th) {
121     String errMsg=getErrMessage(method, th);
122     LOGGER.log(Level.SEVERE, errMsg);
123   }
124   
125   /**
126    * get the resource for the given path
127    * 
128    * @param path
129    * @return
130    * @throws Throwable
131    */
132   public WikiTemplateResource getResource(String path) throws Throwable {
133     if (resources.containsKey(path)) {
134       return resources.get(path);
135     }
136     // this is a new resource - check whether we need to load it from the wiki
137     if (path.startsWith("wiki.")) {
138       LOGGER.log(Level.INFO,"loading WikiTemplateResource from "+path);
139       WikiTemplateResource wikiTemplate = new WikiTemplateResource(this, path,
140           wiki);
141       if (wikiTemplate.rythmTemplateCode!=null)
142         resources.put(path, wikiTemplate);
143       return wikiTemplate;
144     }
145     return null;
146   }
147   
148 
149   @Override
150   public String getResourceLoaderRoot() {
151     String result = "wiki";
152     return result;
153   }
154 
155   @Override
156   public ITemplateResource load(String path) {
157     WikiTemplateResource result=null;
158     try {
159       result = getResource(path);
160     } catch (Throwable th) {
161       handle("load ",th);
162       return new WikiTemplateResource(this,th);
163     }
164     return result;
165   }
166 
167   @Override
168   public void scan(TemplateResourceManager arg0) {
169     // do nothing
170   }
171 
172   /**
173    * register the given Template
174    * @param key
175    * @param wikiTemplate
176    * @return
177    */
178   public TemplateClass registerTemplate(String key,WikiTemplateResource wikiTemplate ) {
179     TemplateClass tc = new TemplateClass(wikiTemplate, engine);
180     ITemplate template = tc.asTemplate(engine);
181     if (null != template) {
182       engine.registerTemplate(key, template);
183     }
184     return tc;
185   }
186   
187   @Override
188   public TemplateClass tryLoadTemplate(String tmplName, RythmEngine engine,
189       TemplateClass callerClass, ICodeType codeType) {
190     if (engine.templateRegistered(tmplName))
191       return null;
192     if (!tmplName.startsWith("wiki."))
193       return null;
194     String rythmSuffix = engine.conf().resourceNameSuffix();
195     String key = tmplName;
196     LOGGER.log(Level.INFO, "Loading wiki template '" + tmplName
197         + "' for rythmSuffix " + rythmSuffix);
198     TemplateClass tc = engine.classes().getByTemplate(key);
199     if (tc == null) {
200       try {
201         WikiTemplateResource wikiTemplate = getResource(key);
202         if (wikiTemplate == null) {
203           throw new IllegalArgumentException("Template for " + key
204               + " not found");
205         }
206         tc=registerTemplate(key,wikiTemplate);
207       } catch (Throwable th) {
208         handle("tryLoadTemplate", th);
209         tc=registerTemplate(key,new WikiTemplateResource(this,th));
210       }
211     }
212     return tc;
213   }
